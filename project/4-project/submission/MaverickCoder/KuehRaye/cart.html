// Get references to the elements var addToCartButton = document.getElementById('add-to-cart'); var quantityInput = document.getElementById('quantity'); var priceElement = document.getElementById('price'); var totalElement = document.getElementById('total'); var subtotalElement = document.getElementById('subtotal'); var shippingInput = document.getElementById('shipping'); var totalPriceElement = document.getElementById('total-price'); // Initialize variables var price = parseFloat(priceElement.textContent); var quantity = parseInt(quantityInput.value); var shipping = parseFloat(shippingInput.value); // Add click event listener to the "Add to Cart" button addToCartButton.addEventListener('click', function() { // Update quantity and calculate total quantity = parseInt(quantityInput.value); var total = price * quantity; // Update total text totalElement.textContent = total.toFixed(2); // Calculate subtotal and total price var subtotal = parseFloat(subtotalElement.textContent); subtotal += total; var totalPrice = subtotal + shipping; // Update subtotal and total price text subtotalElement.textContent = subtotal.toFixed(2); totalPriceElement.textContent = totalPrice.toFixed(2); // Display a success message alert('Item added to cart!'); }); // Update total price when shipping value changes shippingInput.addEventListener('input', function() { shipping = parseFloat(shippingInput.value); var subtotal = parseFloat(subtotalElement.textContent); var totalPrice = subtotal + shipping; totalPriceElement.textContent = totalPrice.toFixed(2); });
